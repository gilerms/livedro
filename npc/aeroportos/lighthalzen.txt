lhz_airport,143,43,5	script	Equipe do Aeroporto#lhz_air1a::lhz_airport1	90,{
	mes "[Equipe do Aeroporto]";
	mes "Bem vindo ao";
	mes "Aeroporto de Lighthalzen";
	mes "onde oferecemos frequentes";
	mes "vôos para Einbroch, Juno e Hugel.";
	next;
	if (select("Embarque.:Cancelar.") == 1) {
		mes "[Equipe do Aeroporto]";
		mes "A taxa do voo é de";
		mes "1,200 zeny, mas caso você";
		mes "possua o Bilhete para o Aeroplano,";
		mes "essa taxa não é cobrada.";
		next;
		if (select("Sim:Não") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Bilhete para o Aeroplano
				warp "lhz_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "lhz_airport",148,51;
				end;
			}
			mes "[Equipe do Aeroporto]";
			mes "Desculpe, mas você não possui";
			mes "os 1,200 zenys para pagar";
			mes "o seu voo.";
			close;
		}
	}
	mes "[Equipe do Aeroporto]";
	mes "Obrigado, e";
	mes "por favor, volte novamente.";
	mes "Tenha um bom dia~";
	close;
}
